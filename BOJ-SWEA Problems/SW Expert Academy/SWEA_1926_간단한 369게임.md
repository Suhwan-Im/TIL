# SWEA #1926 - 간단한 369게임



### #1차 - 220202

```python
### 입력
N = int(input())

### 계산
# 정답을 담을 빈 리스트 생성
list_369 = []

# for문을 이용해서 3, 6, 9가 들어가는 경우 '-'로 출력
for num in range(1, N+1):
    # 네자리수
    if len(str(num)) == 4:
        list_369.append(str(num))
    
    # 세자리수
    elif len(str(num)) == 3:
        if str(num)[0] in ['3', '6', '9']:
            if str(num)[1] in ['3', '6', '9']:
                if str(num)[2] in ['3', '6', '9']:
                    list_369.append('---')
                else:
                    list_369.append('--')
            elif str(num)[2] in ['3', '6', '9']:
                list_369.append('--')
            else:
                list_369.append('-')
        elif str(num)[1] in ['3', '6', '9']:
            if str(num)[2] in ['3', '6', '9']:
                list_369.append('--')
            else:
                list_369.append('-')
        elif str(num)[2] in ['3', '6', '9']:
            list_369.append('-')
        else:
            list_369.append(str(num))
    
    # 두자리수
    elif len(str(num)) == 2:
        if str(num)[0] in ['3', '6', '9']:
            if str(num)[1] in ['3', '6', '9']:
                list_369.append('--')
            else:
                list_369.append('-')
        elif str(num)[1] in ['3', '6', '9']:
            list_369.append('-')
        else:
            list_369.append(str(num))
    
    # 한자리수
    else:
        if str(num)[0] in ['3', '6', '9']:
            list_369.append('-')
        else:
            list_369.append(str(num))

### 출력
print(' '.join(list_369))
```
